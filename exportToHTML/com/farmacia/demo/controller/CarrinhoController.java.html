1    package com.farmacia.demo.controller;
2    
3    import com.farmacia.demo.enums.FormaPagamento;
4    import com.farmacia.demo.model.CarrinhoDeCompras;
5    import com.farmacia.demo.model.Carrinho;
6    import com.farmacia.demo.model.FinalizacaoCompra;
7    import org.springframework.web.bind.annotation.*;
8    
9    @RestController
10   @RequestMapping("/carrinho")
11   public class CarrinhoController {
12   
13       // Simulação: Injetando o carrinho. No mundo real, você usaria um Service
14       // e o escopo da sessão para gerenciar o carrinho de cada usuário.
15       private final CarrinhoDeCompras carrinho;
16   
17       public CarrinhoController() {
18           this.carrinho = new CarrinhoDeCompras();
19       }
20   
21       // POST /carrinho/adicionar
22       // Adiciona um novo item ao carrinho.
23       @PostMapping("/adicionar")
24       public CarrinhoDeCompras adicionarItem(@RequestBody Carrinho item) {
25           carrinho.adicionarItem(item);
26           return carrinho; // Retorna o estado atual do carrinho
27       }
28   
29       // GET /carrinho/total
30       // Retorna o valor total da compra.
31       @GetMapping("/total")
32       public double getTotal() {
33           return carrinho.calcularValorTotal();
34       }
35   
36       // GET /carrinho/info
37       // Retorna todos os detalhes do carrinho.
38       @GetMapping("/info")
39       public CarrinhoDeCompras getCarrinho() {
40           return carrinho;
41       }
42   
43       // POST /carrinho/pagamento?forma=PIX
44       // Escolhe a forma de pagamento.
45       @PostMapping("/pagamento")
46       public String escolherPagamento(@RequestParam FormaPagamento forma) {
47           carrinho.escolherFormaPagamento(forma);
48           double total = carrinho.calcularValorTotal();
49           return String.format("Forma de pagamento escolhida: %s. Valor total: R$ %.2f", forma.name(), total);
50       }
51   
52       @PostMapping("/finalizar")
53       public String finalizarCompra(@RequestBody FinalizacaoCompra finalizacaoInfo) {
54   
55           // 1. Pré-validação do Carrinho
56           double total = carrinho.calcularValorTotal();
57   
58           if (carrinho.getItens().isEmpty()) {
59               return "Erro: O carrinho está vazio. Adicione itens antes de finalizar.";
60           }
61   
62           if (carrinho.getFormaPagamento() == null) {
63               return "Erro: Selecione uma forma de pagamento antes de finalizar.";
64           }
65   
66           // 2. Simulação de Confirmação de Login
67   
68           // Simulação de validação de nome obrigatório
69           if (finalizacaoInfo.getNome() == null || finalizacaoInfo.getNome().isEmpty()) {
70               return "Erro: O nome do cliente é obrigatório para finalizar a compra.";
71           }
72   
73           // Simulação de validação de senha:
74           // *SUBSTITUA "sua-senha-segura" pela sua lógica real de login e hashing de senha.*
75           if (!finalizacaoInfo.getSenha().equals("sua-senha-segura")) {
76               return "Erro: Senha inválida para o cliente " + finalizacaoInfo.getNome();
77           }
78   
79           // 3. Processamento da Transação
80   
81           String recibo = String.format(
82                   "Compra #12345 finalizada com sucesso!\nCliente: %s\nTotal Pago: R$ %.2f\nForma de Pagamento: %s\nObrigado por comprar!",
83                   finalizacaoInfo.getNome(),
84                   total,
85                   carrinho.getFormaPagamento().name()
86           );
87   
88           // 4. Limpar o carrinho após a finalização
89           carrinho.limparCarrinho();
90   
91           return recibo;
92       }
93   
94   }
